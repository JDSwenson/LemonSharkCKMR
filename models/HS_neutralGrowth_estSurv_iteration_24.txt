model {
  Nf ~ dnorm(0, 1e-06)
  Nm ~ dnorm(0, 1e-06)
  surv ~ dbeta(1, 1)
  for (i in 1:mom_yrs) {
    MHSP[i] ~ dbin((surv^(mom_ys_birth[i] - mom_os_birth[i]))/(Nf * lam^(mom_ys_birth[i] - min_cohort)), mom_n_comps[i])
  }
  for (j in 1:dad_yrs) {
    FHSP[j] ~ dbin((surv^(dad_ys_birth[j] - dad_os_birth[j]))/(Nm * lam^(dad_ys_birth[j] - min_cohort)), dad_n_comps[j])
  }
}
